Debug Scroll Issue

Debug Script - Add this to your page temporarily

Add this script tag at the VERY END of your HTML body, right before </body>:

<script>
// TEMPORARY DEBUG SCRIPT - REMOVE AFTER FIXING
(function() {
    console.log('🔍 DEBUG: Monitoring all scroll events...');
    
    // Override scrollIntoView globally
    const originalScrollIntoView = Element.prototype.scrollIntoView;
    Element.prototype.scrollIntoView = function(...args) {
        console.error('🚨 scrollIntoView called on:', this);
        console.trace('Stack trace:');
        // Comment out the next line to prevent scrolling entirely
        // originalScrollIntoView.apply(this, args);
    };
    
    // Monitor scroll events
    let lastScrollY = window.scrollY;
    window.addEventListener('scroll', function(e) {
        const delta = window.scrollY - lastScrollY;
        if (Math.abs(delta) > 100) {
            console.warn('🚨 Large scroll detected:', delta, 'pixels');
            console.trace('Stack trace:');
        }
        lastScrollY = window.scrollY;
    }, true);
    
    // Monitor focus events (focus can cause scrolling)
    document.addEventListener('focus', function(e) {
        console.log('📍 Focus event on:', e.target);
    }, true);
    
    // Check for forms
    document.addEventListener('DOMContentLoaded', function() {
        const forms = document.querySelectorAll('form');
        if (forms.length > 0) {
            console.warn('⚠️ Found', forms.length, 'form(s) on page');
            forms.forEach((form, i) => {
                console.log(`Form ${i}:`, form);
                form.addEventListener('submit', function(e) {
                    console.error('🚨 Form submission detected!');
                    e.preventDefault();
                    return false;
                });
            });
        }
        
        // Check search elements
        const searchInput = document.getElementById('searchInput');
        const searchButton = document.getElementById('searchButton');
        const searchResults = document.getElementById('searchResults');
        
        console.log('Search elements found:');
        console.log('- Input:', !!searchInput);
        console.log('- Button:', !!searchButton);
        console.log('- Results:', !!searchResults);
        
        if (searchInput) {
            // Check if input is inside a form
            const form = searchInput.closest('form');
            if (form) {
                console.error('🚨 Search input is inside a form!', form);
            }
        }
    });
})();
</script>

What this does:

  • Intercepts ALL scrollIntoView calls and logs them
  • Monitors large scroll movements
  • Detects form submissions
  • Tracks focus events
  • Checks if search is wrapped in a form

How to use:

  1. Copy the script above
  2. Add it to the END of your HTML files (both characters_hub.html and women-bible-hub.html)
  3. Open browser console (F12)
  4. Search for "Boaz"
  5. Look for red error messages showing what's causing the scroll
  6. Share the console output
